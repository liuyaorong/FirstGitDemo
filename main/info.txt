Git 使用的常用命令：

git init	#初始化本地版本库
git add	XXX         #将XXX文件添加到暂存区
git commit -m "修改的内容描述"  #提交修改操作，此时会将修改的内容提交到版本库中
git status          #查看仓库的状态
git diff XXX        #查看XXX文件修改了哪些内容
git log             #查看历史记录
git reset --hard HEAD^或者git reset --hard HEAD~  #回退到上一个版本
cat XXX             #查看XXX文件的内容
git reflog          #查看历史记录的版本号ID
git checkout --XXX  #把XXX文件在工作区的修改全部撤销
git rm XXX          #删除XXX文件


git checkout -b dev  #创建dev分支 并切换dev分支上
git branch           #查看当前所有的分支
git checkout master  #切换到master分支
git merge dev        #在当前的分支上合并dev分支
git branch -d dev    #删除dev分支
git stash            #把当前的工作隐藏起来，等以后需要恢复的时候继续工作
git stash list       #查看所有被隐藏的文件列表
git stash apply      #恢复被隐藏的文件，但是内容不删除
git stash drop       #删除文件
git stash pop        #恢复文件的同时，也删除文件
git remote           #查看远程仓库的信息
git remote -v        #查看远程仓库的详细信息


git remote add origin https://github.com/liuyaorong/远程版本库名称.git		#关联一个远程仓库  
git clone https://github.com/liuyaorong/远程版本库名称.git			#从远程仓库中克隆到本地
git push -u(仅限第一次使用-u) origin 版本库名称(master)				#Git会把该分支推送到远程仓库对应的远程分支上
